function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); // ── SORIANA ── if (data.action === 'syncSoriana') { const sheet = ss.getSheetByName('Soriana') || ss.insertSheet('Soriana'); const HEADERS = ['id','factura','pedido','numTienda','sucursal','monto','mes', 'portal','miro','numMiro','fechaMiro','migo','numMigo','fechaMigo', 'fechaCompromiso','fechaCobro','montoCobrado','categoria', 'canceladaMigo','canceladaSinServicio','duplicada', 'comentarios','calendario','updatedAt']; if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); // Reemplazar todo con los datos recibidos (fuente de verdad = app) const existing = sheet.getLastRow(); if (existing > 1) sheet.deleteRows(2, existing - 1); const rows = (data.facturas||[]).map(f => HEADERS.map(h => { const v = f[h]; return (v === true) ? true : (v === false) ? false : (v ?? ''); })); if (rows.length > 0) sheet.getRange(2, 1, rows.length, HEADERS.length).setValues(rows); return ContentService.createTextOutput(JSON.stringify({ok:true,saved:rows.length})).setMimeType(ContentService.MimeType.JSON); } // ── CARTERA VENCIDA ── if (data.action === 'syncCartera') { const sheet = ss.getSheetByName('Cartera') || ss.insertSheet('Cartera'); const HEADERS = ['id','cliente','vigente','b1_30','b31_60','b61_90','bMas90', 'totalVencido','total','gestionJSON']; if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); const existing = sheet.getLastRow(); if (existing > 1) sheet.deleteRows(2, existing - 1); const rows = (data.clientes||[]).map(c => [ c.id, c.cliente, c.vigente, c.b1_30, c.b31_60, c.b61_90, c.bMas90, c.totalVencido, c.total, JSON.stringify(c.gestion||null) ]); if (rows.length > 0) sheet.getRange(2, 1, rows.length, HEADERS.length).setValues(rows); return ContentService.createTextOutput(JSON.stringify({ok:true,saved:rows.length})).setMimeType(ContentService.MimeType.JSON); } // ── HISTÓRICO DE CARTERA (snapshots por fecha de carga) ── if (data.action === 'syncCarteraHistorico') { const sheet = ss.getSheetByName('CarteraHistorico') || ss.insertSheet('CarteraHistorico'); if (sheet.getLastRow() === 0) sheet.appendRow(['historicoJSON', 'updatedAt']); if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow() - 1); sheet.getRange(2, 1, 1, 2).setValues([[JSON.stringify(data.historico||{}), new Date().toISOString()]]); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } // ── PROMESAS ── if (data.action === 'savePromesa') { const sheet = ss.getSheetByName('Promesas') || ss.insertSheet('Promesas'); if (sheet.getLastRow() === 0) { sheet.appendRow(['id','cliente','fecha','monto','persona','estado', 'historial','incumplimientos','montoReal','cerradoPor','createdAt']); } sheet.appendRow([ data.id, data.cliente, data.fecha, data.monto, data.persona, data.estado, JSON.stringify(data.historial||[]), data.incumplimientos||0, data.montoReal||0, data.cerradoPor||'', data.createdAt||new Date().toISOString() ]); } else if (data.action === 'updatePromesa') { const sheet = ss.getSheetByName('Promesas'); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { sheet.getRange(i+1, 4).setValue(data.monto); sheet.getRange(i+1, 6).setValue(data.estado); sheet.getRange(i+1, 7).setValue(JSON.stringify(data.historial||[])); sheet.getRange(i+1, 8).setValue(data.incumplimientos||0); sheet.getRange(i+1, 9).setValue(data.montoReal||0); sheet.getRange(i+1, 10).setValue(data.cerradoPor||''); if (data.fecha) sheet.getRange(i+1, 3).setValue(data.fecha); break; } } } else if (data.action === 'deletePromesa') { const sheet = ss.getSheetByName('Promesas'); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { sheet.deleteRow(i+1); break; } } } else { // ── REGISTROS DIARIOS ── const sheet = ss.getSheetByName('Registros') || ss.getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['fecha','hora','persona','llamadas','monto', 'cobrado','escalados','comentario','programaciones','id']); } sheet.appendRow([ data.fecha, data.hora, data.persona, data.llamadas, data.monto, data.cobrado, data.escalados, data.comentario, JSON.stringify(data.programaciones||[]), data.id||'' ]); } return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const result = {}; // Registros diarios const regSheet = ss.getSheetByName('Registros') || ss.getActiveSheet(); const regRows = regSheet.getDataRange().getValues(); if (regRows.length > 1) { const headers = regRows[0]; result.registros = regRows.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); try { obj.programaciones = JSON.parse(obj.programaciones||'[]'); } catch(e) { obj.programaciones = []; } return obj; }); } else { result.registros = []; } // Promesas const promSheet = ss.getSheetByName('Promesas'); if (promSheet) { const promRows = promSheet.getDataRange().getValues(); if (promRows.length > 1) { const ph = promRows[0]; result.promesas = promRows.slice(1).map(r => { const obj = {}; ph.forEach((h,i) => obj[h] = r[i]); try { obj.historial = JSON.parse(obj.historial||'[]'); } catch(e) { obj.historial = []; } obj.incumplimientos = Number(obj.incumplimientos)||0; obj.montoReal = Number(obj.montoReal)||0; obj.monto = Number(obj.monto)||0; return obj; }); } else { result.promesas = []; } } else { result.promesas = []; } // Soriana const sorSheet = ss.getSheetByName('Soriana'); if (sorSheet && sorSheet.getLastRow() > 1) { const sh = sorSheet.getDataRange().getValues(); const headers = sh[0]; // Campos que son fechas y deben salir como YYYY-MM-DD, no como Date objects const dateFields = new Set(['fechaMiro','fechaMigo','fechaCompromiso','fechaCobro','updatedAt']); result.soriana = sh.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => { let v = r[i]; if (v === 'true') v = true; else if (v === 'false') v = false; else if (dateFields.has(h) && v instanceof Date) { // Convertir Date de Sheets a YYYY-MM-DD sin problema de zona horaria const yy = v.getFullYear(); const mm = String(v.getMonth()+1).padStart(2,'0'); const dd = String(v.getDate()).padStart(2,'0'); v = `${yy}-${mm}-${dd}`; } else if (dateFields.has(h) && typeof v === 'string' && v.includes('T')) { // Si llegó como ISO string con hora, quedarse solo con la fecha v = v.substring(0,10); } obj[h] = v ?? ''; }); obj.monto = Number(obj.monto)||0; obj.montoCobrado = Number(obj.montoCobrado)||0; // Asegurar que mes sea string limpio '2026-06', no objeto Date if (obj.mes instanceof Date) { obj.mes = obj.mes.getFullYear()+'-'+String(obj.mes.getMonth()+1).padStart(2,'0'); } else if (typeof obj.mes === 'string' && obj.mes.includes('T')) { obj.mes = obj.mes.substring(0,7); } return obj; }); } else { result.soriana = []; } // Cartera vencida const carteraSheet = ss.getSheetByName('Cartera'); if (carteraSheet && carteraSheet.getLastRow() > 1) { const ch = carteraSheet.getDataRange().getValues(); const cHeaders = ch[0]; result.cartera = ch.slice(1).map(r => { const obj = {}; cHeaders.forEach((h,i) => { obj[h] = r[i] ?? ''; }); obj.vigente = Number(obj.vigente)||0; obj.b1_30 = Number(obj.b1_30)||0; obj.b31_60 = Number(obj.b31_60)||0; obj.b61_90 = Number(obj.b61_90)||0; obj.bMas90 = Number(obj.bMas90)||0; obj.totalVencido = Number(obj.totalVencido)||0; obj.total = Number(obj.total)||0; try { obj.gestion = obj.gestionJSON ? JSON.parse(obj.gestionJSON) : null; } catch(e) { obj.gestion = null; } delete obj.gestionJSON; return obj; }); } else { result.cartera = []; } // Histórico de cartera (snapshots) const histSheet = ss.getSheetByName('CarteraHistorico'); if (histSheet && histSheet.getLastRow() > 1) { const hRow = histSheet.getRange(2, 1, 1, 1).getValues()[0][0]; try { result.carteraHistorico = JSON.parse(hRow || '{}'); } catch(e) { result.carteraHistorico = {}; } } else { result.carteraHistorico = {}; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
